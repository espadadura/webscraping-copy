# create project folder
$ mkdir myproject   
#set directory      
$ cd myproject       
                        
# create new env. -- dsp = name of environment
$ conda create –n dsp python=3.6 keras scikit-learn nltk pandas tensorflow jupyterlab

# activate environment
$ activate dsp

# deactivate environment
$ deactivate 

#delete environment
$ conda remove --name dsp --all 

#export environment
[dsp] $ conda env export > environment.yaml

#use exported environment
$ conda env create -f environment.yaml
